// . WAP to Reverse String "good morning I am from Chiplun". output/ "Chiplun from am I morning good" 
public class Main
{
	public static void main(String[] args) {
		String str="good morning I am from Chiplun";
		String words[]=str.split(" ");
		for(int i=words.length-1;i>=0;i--)
		{
		    System.out.print(words[i]+" ");
		}
		System.out.print(" ");
	}
}
